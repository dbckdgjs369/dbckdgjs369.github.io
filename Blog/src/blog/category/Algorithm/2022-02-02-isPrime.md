## 알고리즘 강의

### 에라스토테네스의 체

    function get_primes(num) {
    const prime = [false, false, ...Array(num - 1).fill(true)];

    for (let i = 2; i * i <= num; i++) {
        if (prime[i]) {
        for (let j = i * 2; j <= num; j += i) {
            prime[j] = false;
        }
        }
    }

    return prime.filter(Boolean);
    }

    console.log(get_primes(10).length);
