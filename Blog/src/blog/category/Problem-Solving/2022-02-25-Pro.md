첫번째 풀이

     function solution(s) {
         let answer = 0;
         let i= isCorrect(s);
         for(let i=0; i<s.length;i++){
             let temp= s.slice(i,s.length);
             let add= s.slice(0,i);
             let result=temp+add;
             if(isCorrect(result)){
                 answer++;
            }
         }
         return answer;
     }
     function isCorrect(s){
         let arr1=[["[","]"],["{","}"],["(",")"]];
         let arr=[0,0,0];
         let count=0;
         for(let i=0; i<s.length;i++){
             for(let j=0;j<3;j++){
                 if(arr[j]<0){
                     return false;
                 }else if(s[i]===arr1[j][0]){
                     arr[j]++;
                 }else if(s[i]===arr1[j][1]){
                     arr[j]--;
                 }
             }
         }
         for(let i=0; i<3;i++){
             if(arr[i]!==0){
                 return false;
             }
         }
         return true;
     }

최종 풀이

    function solution(s) {
        let answer = 0;
        for(let i=0; i<s.length;i++){
            let result= s.slice(i,s.length)+s.slice(0,i);
            if(isCorrect(result)){
                answer++;
            }
        }
        return answer;
    }
    function isCorrect(s){
        let arr1=[["[","]"],["{","}"],["(",")"]];
        let stack=[];
        stack.push(s[0])
        for(let i=1;i<s.length;i++){
            for(let j=0; j<3;j++){
                if(s[i]===arr1[j][0]){
                    stack.push(arr1[j][0]);
                }else if(s[i]===arr1[j][1]){
                    if(stack[stack.length-1]===arr1[j][0]){
                        stack.pop()
                    }
                }
            }
        }
        return stack.length===0?true:false;
    }
