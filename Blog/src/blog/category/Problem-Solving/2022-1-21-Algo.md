## 프로그래머스

### Lv 2. 예상 대진표

    function solution(n,a,b)
    {
        var max= Math.max(a,b);
        var min= Math.min(a,b);
        var answer = 0;
        for(let i=0; i<=n/2;i++){
            min=cal(min);
            max=cal(max);
            if(min===max){
                answer++;
                break;
            }else{
                answer++;
            }
        }
        return answer;
    }

    function cal(num){
        if(num%2==0){
            return num/=2;
        }else{
            return num=(num+1)/2;
        }
    }
